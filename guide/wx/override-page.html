export const initGlobalLog = function(){
  let originPage = Page
  Page = function(ctx={}){
    let originHide = ctx.onHide
    let originShow = ctx.onShow
    ctx.onShow = function(){
      onPageShow(ctx)
      originShow && originShow.call(this)
    }
    ctx.onHide = function(){
       onPageHide(ctx)
       originHide && originHide.call(this)
     }
    return originPage(ctx)
  }
}
