# el-table 多表格 分页 多选 mixin


/**
 * 表格单行 点击切换选中事件
 * <el-table ref="tableRef" @row-click="(row)=>this.$refs[tableRef].toggleRowSelection(row)" />
 * 表格全选切换 事件
 * <el-table ref="rableRef" @selection-change="onSelectionChange($event,tableKey,page)" />
 *
 * toggleRowSelection({ resList, page, tableKey, tableRef, idKey })
 */
export default {
  data() {
    return {
      tableSelection: {} // {'tableKey':{page:[]}}
    }
  },
  methods: {
    // 记录 表格 多选
    onSelectionChange(val, tableKey, page) {
      // 使用防抖
      clearTimeout(this.selectionTimer)
      this.selectionTimer = setTimeout(() => {
        clearTimeout(this.selectionTimer)
        if (!this.tableSelection[tableKey]) {
          this.$set(this.tableSelection, tableKey, {})
        }
        this.tableSelection[tableKey][page] = val
      }, 50)
    },
    /**
     * 跨分页 多选
     * @param {Object} resList 请求到的 列表数据
     */
    toggleRowSelection({ resList, page, tableKey, tableRef, idKey }) {
      if (!tableRef) tableRef = tableKey
      page = +page
      if (!this.tableSelection[tableKey]) {
        this.$set(this.tableSelection, tableKey, {})
      }
      const rows = this.tableSelection[tableKey][page] || []
      this.$nextTick(() => {
        if (rows.length) {
          this.tableSelection[tableKey][page] = resList.filter(row => {
            const i = rows.findIndex(v => v[idKey] === row[idKey])
            if (i >= 0) {
              this.$refs[tableRef].toggleRowSelection(row)
            }
            return i >= 0
          })
        } else {
          this.$refs[tableRef].clearSelection()
        }
      })
    },
    /**
     * 重置表格多选
     * @param {Array} tables
     */
    resetTableSelection(tables = []) {
      if (!tables.length) { // 默认全清空
        Object.keys(this.tableSelection).forEach(table => {
          this.tableSelection[table] = {}
        })
      } else {
        tables.forEach(table => { // 清空指定 table 多选
          this.tableSelection[table] = {}
        })
      }
    },
    /**
     * 获取 表格分页多选 组合成一个数组
     * @param {String} tableKey  table key
     */
    getTableSelection(tableKey) {
      let result = []
      const table = this.tableSelection[tableKey] || {}
      Object.keys(table).forEach(page => {
        result = result.concat(table[page] || [])
      })
      return result
    }
  }
}
